      vibeKeywords = ['조용한', '혼밥 가능', '빠른 회전', '1인석'];
    }

    let timeConstraint = '';
    if (state.meal === '아침') timeConstraint = '아침 영업 필수 (7~10시 운영)';

    // ── 8. 인원 관련 조건
    let groupConstraint = '';
    if (n === 1) groupConstraint = '혼밥 가능한 구조 (카운터석 또는 1인 테이블)';
    else if (n >= 5) groupConstraint = `단체석 ${n}명 이상 수용 가능한 곳`;
    else groupConstraint = `${n}명이 편하게 앉을 수 있는 테이블`;

    // ── 9. 2차 카테고리 결정
    let secondRoundCats = [];
    if (state.companion === '고객사') {

    } else {
      secondRoundCats = ['카페', '디저트 카페'];
    }

    return { topCats, vibeKeywords, timeConstraint, groupConstraint, secondRoundCats };
  }

  async function fetchRecommendations(round = 1) {

      const res = await fetch('https://api.anthropic.com/v1/messages', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          model: 'claude-sonnet-4-20250514',
          max_tokens: 1200,
          messages: [{ role: 'user', content: buildPrompt(round) }]
        })
      });

      const data = await res.json();
      const text = data.content.map(b => b.text || '').join('');
      const clean = text.replace(/```json|```/g, '').trim();
      const restaurants = JSON.parse(clean);

      if (round === 1) state.firstRoundNames = restaurants.map(r => r.name);
      showResults(restaurants, round);
    } catch (e) {
      showResults(getFallbackData(round), round);
    }
  }

  function showResults(restaurants, round = 1) {
    document.getElementById('loadingSection').classList.add('hidden');
    document.getElementById('resultsSection').classList.remove('hidden');

    // Tags (only on first round)
    if (round === 1) {
      const tagData = [
        { label: state.meal, emoji: state.meal === '아침' ? '🌅' : state.meal === '점심' ? '☀️' : '🌙' },
        { label: `${state.count}명`, emoji: '👥' },
        { label: state.companion, emoji: '🤝' },
        { label: state.price, emoji: '💰' }
      ];
      document.getElementById('tagSummary').innerHTML = tagData.map(t =>
        `<span class="tag">${t.emoji} ${t.label}</span>`
      ).join('');
      document.getElementById('locationText').textContent = state.locationName || '현재 위치 기준';
      document.getElementById('locationSub').textContent = state.locationDetail || '주변 도보 10분 이내';
      document.getElementById('resultsTitle').textContent = '🎉 추천 맛집이에요!';
      document.getElementById('resultsSub').textContent = `${state.locationName || '현재 위치'} 주변 맛집 3곳`;
    } else {
      document.getElementById('resultsTitle').textContent = '🍻 2차 추천이에요!';
      document.getElementById('resultsSub').textContent = '식사 후 이어가기 좋은 곳';
    }

    const list = document.getElementById('restaurantList');
    const cards = restaurants.slice(0, 3).map((r, i) => `
      <div class="restaurant-card" style="--i: ${i}">
        <div class="r-header">
          <div class="r-rank ${i === 0 ? 'top' : ''}">${i + 1}</div>
          <div class="r-info">
            <div class="r-name">${r.name}</div>
            <div class="r-category">${r.category}</div>
            <div class="r-meta">
              <span class="r-rating">★ ${r.rating}</span>
              <span class="r-distance">📍 ${r.distance}</span>
              <span class="r-price-badge">${r.pricePerPerson}/인</span>
            </div>
          </div>
        </div>
        <div class="r-body">
          <div class="r-desc">💬 ${r.recommend || r.description}</div>
          <div class="r-parking ${r.parking ? 'ok' : 'no'}">
            ${r.parking ? '🅿️' : '🚫'} ${r.parkingNote || (r.parking ? '주차 가능' : '주차 불가')}
          </div>
          <div class="r-tags">
            ${(r.tags || []).map(t => `<span class="r-tag">#${t}</span>`).join('')}
          </div>
        </div>

          <button class="r-action" onclick="openMap('${r.name}')">🗺️ 지도 보기</button>
          <button class="r-action" onclick="searchMore('${r.name}')">🔍 더 알아보기</button>
        </div>
      </div>
    `).join('');

    if (round === 1) {
      list.innerHTML = cards;
      // Show 2nd round button
      document.getElementById('secondRoundBtn').classList.remove('hidden');
      document.getElementById('secondRoundSection').classList.add('hidden');

      `;
      document.getElementById('secondRoundBtn').classList.add('hidden');

  function openMap(name) {
    const q = encodeURIComponent((state.locationName || '') + ' ' + name);
    window.open(`https://map.naver.com/v5/search/${q}`, '_blank');
  }

  function searchMore(name) {
    const q = encodeURIComponent((state.locationName || '') + ' ' + name);
    window.open(`https://search.naver.com/search.naver?query=${q}`, '_blank');
  }
